# Detection: Brute Force Login Attempts
# MITRE ATT&CK: T1110 - Brute Force
# Tactic: Credential Access
# Event ID: 4625 - Failed Logon
# Severity: HIGH
# Description: Detects 5+ failed login attempts from same host within 5 minutes

index=windows EventCode=4625
| stats count by src_ip, Account_Name, host
| where count >= 5
| eval severity="HIGH"
| eval mitre_technique="T1110 - Brute Force"
| eval kill_chain_stage="Credential Access"
| table _time, src_ip, Account_Name, host, count, severity, mitre_technique, kill_chain_stage
| sort - count
